Cutscene Manager Originally the video clips would just play and the player would have no ability to skip or pause them.
So I implemented the functionality to do so. The two main functions are PlayCutscene() and SkipCutscene().

To play a cut scene you add the file to the cutscene manager and input its name.
 [code lang = "csharp"] public void PlayCutscene(string _cutsceneName) { videoPlayer.clip = null; SkipTimer = 1; for (int i = 0; i < CutsceneDatas.Length; i++) { if (CutsceneDatas[i].Clip.name == _cutsceneName) { cutsceneDataInUse = CutsceneDatas[i]; videoPlayer.clip = cutsceneDataInUse.Clip; } } if (videoPlayer.clip != null) { clipLength = videoPlayer.clip.length; RenderTexture rt = RenderTexture.active; RenderTexture.active = videoImage; GL.Clear(true, true, Color.clear); RenderTexture.active = rt; videoCanvas.SetActive(true); videoPlayer.Play(); if (GameManagement.Instance.GetSettingValue("SubtitlesState") == 1 ? true : false) { StartCoroutine(SubtitleManager.PlaySubtitles(cutsceneDataInUse.SubtitleAsset,0.5f)); } playingCutscene = true; GameManagement.Instance.PlayerCanMove = false; } else { Debug.LogError("No cutscene found: Make sure the correct name is used"); } } [/code]
